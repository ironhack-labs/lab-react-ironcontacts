sortByPopularity = () => {
    this.setState(prevState => {
      const { contacts } = prevState;

      contacts.sort((a, b) => {
        if (a.popularity > b.popularity) return 1;
        else if (a.popularity < b.popularity) return -1;
        return 0
      });
      return { contacts };
    });
  };
